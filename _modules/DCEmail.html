[docs]def send_text_email(msg=None, subject=None, email_from=None, email_to=None) :
    """Sends e-mail.
    """
    dmsg = MIMEText(msg)
    dmsg['Subject'] = str(subject)
    dmsg['From']    = email_from
    dmsg['To']      = email_to
    s = smtplib.SMTP('localhost')
    s.sendmail(email_from, [email_to], dmsg.as_string())
    s.quit()
